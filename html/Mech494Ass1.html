a = [3;2;1];
b = [-1;7;2];
c = [-2;5;6];

xx = [a(1),b(1),c(1)];
yy = [a(2),b(2),c(2)];
zz = [a(3),b(3),c(3)];

LG = c;
u = a-c;
u = u/norm(u);
w = cross(u,b-c);
w = w/norm(w);
v = cross(w,u);
R = [u,v,w];

display('Pose matrix of the local system with respect to the global system')
T = [1,0,0,0;LG,R;]

display('Verification of orthogonality, where the determinant = 1')
orthofR = det(R)


hold on
